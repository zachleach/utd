% -------------------------------------------------- 
% CHAPTERS, SECTIONS, AND SUBSECTIONS --------------
% -------------------------------------------------- 

% section numbering in the margins
\usepackage{titlesec}
\newlength\sectionindent
\setlength\sectionindent{54pt}
\titleformat{\section}
{\normalfont\Large\bfseries}{\llap{\makebox[\sectionindent][l]{\thesection\hfill}}}{0pt}{}
\titlespacing*{\section}{0pt}{8pt}{4pt}

\titleformat{\subsection}
{\normalfont\bfseries}{\llap{\makebox[\sectionindent][l]{\bfseries\thesubsection\hfill}}}{0pt}{}
\titlespacing*{\subsection}{0pt}{8pt}{4pt}

% removed "Chapter XX" text
\titleformat{\chapter}[display]
{\normalfont\huge\bfseries}{}{20pt}{\Huge}
\titlespacing*{\chapter}{0pt}{0pt}{20pt}
